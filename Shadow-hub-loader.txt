local LoaderConfig = {
    Enabled = true,
    SupportedGames = {
        [116756675465096] = "https://raw.githubusercontent.com/aleksmago1/Shadowbypasser/refs/heads/main/obfuscated_script-1770065726533.lua.txt", -- jujutsu slap tower
     },
    NotificationSettings = {
        Title = "Game Loader",
        Duration = 5
    }
}

if not LoaderConfig.Enabled then
    return 
end

local currentPlaceId = game.PlaceId

if LoaderConfig.SupportedGames[currentPlaceId] then
    local success, err = pcall(function()
        loadstring(game:HttpGet(LoaderConfig.SupportedGames[currentPlaceId]))()
    end)
    
    if not success then
        warn("Failed to load game script: " .. err)
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = LoaderConfig.NotificationSettings.Title,
            Text = "Failed to load game script, try to re-execute.",
            Duration = LoaderConfig.NotificationSettings.Duration
        })
    end
else
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = LoaderConfig.NotificationSettings.Title,
        Text = "Game not supported",
        Duration = LoaderConfig.NotificationSettings.Duration
    })
end
